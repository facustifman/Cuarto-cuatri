```C++

class TELEFONO{
    private:
        int area;
        int numero;
    public:
        TELEFONO(int a, int n) : area(a), numero(n) {}

        TELEFONO(const TELEFONO& other) : area(other.area), numero(other.numero) {}//constructor de copia
        TELEFONO(TELEFONO&& other) noexcept : area(other.area), numero(other.numero) { other.area = 0; other.numero = 0;}//constructor de movimiento

            if (this != &other) {
                area = other.area;
                numero = other.numero;

            return *this;

            return (area == other.area && numero == other.numero);
```

5) ¿Qué significa que una función es blocante?¿Cómo subsanaría esa limitación en término de mantener el programa ‘vivo’ ?

Una funcion es bloqueante cuando su ejecucion detiene al resto del programa haciendo imposible que las otras tareas continuen ejecutandose hasta que la funcion termine. Para subsanar esta limitacion y mantener vivo el programa se podria utilizar hilos para ejecutar la funcion bloqueante en paralelo con el resto del programa, permitiendo que otras tareas se ejecuten mientras la funcion bloqueante esta en progreso.


6) Explique qué son los métodos virtuales puros y para qué sirven. De un breve ejemplo donde su uso sea imprescindible.

Los metodos virtuales puros son funciones miembro de una clase a la cual se le puede realizar un Override en una clase derviada para pisar su contenido. Se declaran con la sintaxis = 0 al final de la declaracion de la funcion. Sirven para definir una interfaz que las clases derivadas deben implementar, permitiendo el polimorfismo y la extensibilidad del código.

```C++
class Figura {
    public:
        virtual int area() = 0; // metodo virtual puro
};

class Rectangulo : public Figura {
    private:
        int ancho;
        int alto;
    public:
        Rectangulo(int a, int h) :ancho(a), alto(h) {}
        int area() override {
            return ancho * alto;
        }
};
class circulo : public Figura {
    private:
        int radio;
    public:
        Circulo(int r) : radio(r) {}
        int area() override {
            return 3.14 * radio * radio;
        }
};
```

7) Escribir un programa C que procese el archivo “numeros.txt” sobre sí mismo (sin crear archivos intermedios y sin subir el archivo a memoria). El procesamiento consiste en leer nros
hexadecimales de 4 símbolos y reemplazarlos por su valor decimal (en texto).

```C
int main(){
    FILE *f= fopen("numeros.txt","r+");
    char hex[5];
    int wrpos=0;
    int rdpos=0;
    while(fread(&hex, 4 , 1, f)==1){
        hex[4]='\0';
        int dec = strtol(hex, NULL, 16);
        fseek(f, wrpos, SEEK_SET);
        fwrite(&dec, sizeof(int), 1, f);
        wrpos += sizeof(int);
        rdpos += 4;
        fseek(f, rdpos, SEEK_SET);
    }
    fclose(f);
    return 0;
}

```


8) ¿Qué es el polimorfismo? Ejemplifique mediante código.

El polimorfismo es un principio de la programación orientada a objetos que permite que una función o método pueda tener diferentes comportamientos según el objeto que lo invoque. Esto se logra mediante la herencia y los métodos virtuales.

```C++
class Animal {
    public:
        virtual void hacerSonido() {
            std::cout << "El animal hace un sonido" << std::endl;
        }
        virtual ~Animal() {}
};
class Perro : public Animal {
    public:
        void hacerSonido() override {
            std::cout << "El perro ladra" << std::endl;
        }
        virtual ~Perro() {}
};
class Gato : public Animal {
    public:
        void hacerSonido() override {
            std::cout << "El gato maulla" << std::endl;
        }
        virtual ~Gato() {}
};
int main() {
    Animal* animal1 = new Perro();
    Animal* animal2 = new Gato();
    animal1->hacerSonido(); // Salida: El perro ladra
    animal2->hacerSonido(); // Salida: El gato maulla
    return 0;
}
```


9) ¿Qué función utiliza para esperar la terminación de un thread? Ejemplifique mediante código.

la funcion que se utiliza para esperar la terminacion de un thread en C++ es join(). Esta función bloquea el hilo que la llama hasta que el hilo asociado haya terminado su ejecución.

```C++
#include <iostream>
#include <thread>
int incrementar(int& contador) {
    for (int i = 0; i < 1000; ++i) {
        ++contador;
    }
    return contador;
}
int main() {
    int contador = 0;
    std::thread hilo(incrementar, std::ref(contador));
    hilo.join(); // Espera a que el hilo termine
    std::cout << "Contador final: " << contador << std::endl;
    return 0;
}
```

10)   Escriba un programa C que tome 2 cadenas por línea de comandos: A y B; e imprima la
cadena A después de haber duplicado todas las ocurrencias de B..
ej.: reemp.exe “El final está aprobado” aprobado -----> El final está aprobado aprobado

```C
#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Uso: %s <cadena A> <cadena B>\n", argv[0]);
        return 1;
    }

    char *A = argv[1];
    char *B = argv[2];
    char resultado[1024] = ""; // Buffer para el resultado
    char *pos = A; // Puntero para recorrer la cadena A
    size_t lenB = strlen(B);

    while ((pos = strstr(pos, B)) != NULL) {
        strncat(resultado, A, pos - A); // Copia la parte antes de B
        strcat(resultado, B); // Duplicar B
        strcat(resultado, B);
        pos += lenB; // Avanza el puntero después de B
        A = pos; // Actualiza A para la siguiente iteración
    }
    strcat(resultado, A); // Copia el resto de A

    printf("%s\n", resultado);
    return 0;
}
```
