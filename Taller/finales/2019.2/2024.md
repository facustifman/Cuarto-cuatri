1) Que es una condition Variable? Ejemplifique mediante código.

Una condition variable es una forma de sincronizacion de threads que permite a un thread esperar hasta que se cumpla una condicion especifica. Se utiliza junto a un mutex para proteger el acceso a los datos compartidos y evitar condiciones de carrera.

```C++
std::mutex mtx;
std::condition_variable cv;
bool ready = false; // The predicate variable

void worker_thread() {
    // 1. Acquire the unique lock
    std::unique_lock<std::mutex> lock(mtx);
    
    // 2. Wait until 'ready' becomes true (handles spurious wakeups)
    cv.wait(lock, []{ return ready; });
    
    // 3. Perform work safely under the lock
    std::cout << "Worker is executing!" << std::endl;
}

void publisher_thread() {
    {
        std::lock_guard<std::mutex> lock(mtx);
        ready = true; // Update predicate
    }
    cv.notify_one(); // Wake up the waiting thread
}
int main() {
    std::thread worker(worker_thread);
    std::thread publisher(publisher_thread);
    worker.join();
    publisher.join();
    return 0;
}

```
2) Que se entiende por RAII? Para que se usa? Que ventajas ofrece? Ejemplifique

Raii es un principio de programacion en C++ que se utiliza para gestionar recursos de manera automatica. La idea es que los reccursos que se adquieren en un objeto se liberan automaticamente cuando el objeto sale de su alcance. Esto ayuda a prevenir fugas de memoria y otros problemas relacionados con la gestión de recursos.
```C++
#include <iostream>
class Resource {
    private:
        int x;
        std:: mutex mtx;
    public:
        Resource(int val) : x(val) {
            mtx.lock(); // Acquire resource
        }
        ~Resource() {
            mtx.unlock(); // Release resource
        }
};

```

4) Que es un iterador STL? Para que se usa? Ejemplifique mediante código.

Un iterador STL es un objeto que permite recorrer los elementos de una colección de datos (como vectores, listas, mapas, etc.) de manera uniforme y eficiente. Se utiliza para acceder a los elementos de la colección sin exponer la estructura interna de la misma.

```C++
#include <iostream>
#include <vector>
int main() {
    std::vector<int> numeros = {1, 2, 3, 4, 5};
    for (std::vector<int>::iterator it = numeros.begin(); it != numeros.end(); ++it) {
        std::cout << *it << " "; // Accede al valor del elemento
    }
    return 0;
}
```

5) Declar una clase LISTA para encapsular una lista doblemente enlazada de elementos genericos tipo T implementada con memoria dinamica. Atributos necesarios, constructor MOVE, +, ==, impresino y carga desde consola. Ejemplifique mediante código.

```C++
#include <iostream>

template <typename T>
class Nodo {
public:
    T dato;
    Nodo<T>* siguiente;
    Nodo<T>* anterior;

    Nodo(T val) : dato(val), siguiente(nullptr), anterior(nullptr) {}
};

template <typename T>
class Lista {
private:
    Nodo<T>* cabeza;
    Nodo<T>* actual;
    Nodo<T>* cola;

public:
    Lista() : cabeza(nullptr), actual(nullptr), cola(nullptr) {}

    ~Lista() {
        Nodo<T>* temp = cabeza;
        while (temp) {
            Nodo<T>* sig = temp->siguiente;
            delete temp;
            temp = sig;
        }
    }

    Lista(const Lista<T>& otra) = delete;
    Lista<T>& operator=(const Lista<T>& otra) = delete;

    Lista(Lista<T>&& otra) noexcept
        : cabeza(otra.cabeza), actual(otra.actual), cola(otra.cola) {
        otra.cabeza = nullptr;
        otra.actual = nullptr;
        otra.cola = nullptr;
    }

    Lista<T>& operator=(Lista<T>&& otra) noexcept {
        if (this != &otra) {
            this->~Lista();

            cabeza = otra.cabeza;
            actual = otra.actual;
            cola = otra.cola;

            otra.cabeza = nullptr;
            otra.actual = nullptr;
            otra.cola = nullptr;
        }

        return *this;
    }

    void agregar(T val) {
        Nodo<T>* nuevo = new Nodo<T>(val);

        if (!cabeza) {
            cabeza = nuevo;
            actual = nuevo;
            cola = nuevo;
        } else {
            cola->siguiente = nuevo;
            nuevo->anterior = cola;
            cola = nuevo;
        }
    }

    Nodo<T>* siguiente() {
        if (actual) {
            actual = actual->siguiente;
        }
        return actual;
    }

    Nodo<T>* anterior() {
        if (actual) {
            actual = actual->anterior;
        }
        return actual;
    }

    T& obtenerDato() {
        return actual->dato;
    }

    bool operator==(const Lista<T>& otra) const {
        Nodo<T>* temp1 = cabeza;
        Nodo<T>* temp2 = otra.cabeza;

        while (temp1 && temp2) {
            if (temp1->dato != temp2->dato) {
                return false;
            }

            temp1 = temp1->siguiente;
            temp2 = temp2->siguiente;
        }

        return temp1 == nullptr && temp2 == nullptr;
    }

    friend std::ostream& operator<<(std::ostream& os, const Lista<T>& lista) {
        Nodo<T>* temp = lista.cabeza;

        while (temp) {
            os << temp->dato << " ";
            temp = temp->siguiente;
        }

        return os;
    }

    friend std::istream& operator>>(std::istream& is, Lista<T>& lista) {
        T val;

        while (is >> val) {
            lista.agregar(val);
        }

        return is;
    }
};
```

7) Implemente con ISO C la funcion int atrasar(FILE f, unsigned offset, unsigned offsetfrom, unsigned legth)
que copie a la posicion offsetto del archivo f, delimitado por length bytes desde Offsetfrom. La funcion debe devolver un entero con el siguiente significado: 0 si la operacion fue exitosa, -1 si hubo error de lectura, -2 si hubo error de escritura y -3 si hubo error de posicionamiento.

```C
#include <stdio.h>

int atrasar(FILE *f, unsigned offsetto, unsigned offsetfrom, unsigned length) {
    fseek(f, offsetfrom, SEEK_SET);
    char *buffer = (char *)malloc(length);
    if (buffer == NULL) {
        return -1; // Error de lectura
    }
    while (length > 0){
        size_t bytesRead = fread(buffer, 1, length, f);
        if (bytesRead == 0) {
            free(buffer);
            return -1; // Error de lectura
        }
        fseek(f, offsetto, SEEK_SET);
        size_t bytesWritten = fwrite(buffer, 1, bytesRead, f);
        if (bytesWritten != bytesRead) {
            free(buffer);
            return -2; // Error de escritura
        }
        offsetto += bytesWritten;
        offsetfrom += bytesRead;
        length -= bytesRead;
    }
    free(buffer);
    return 0; // Operación exitosa
}
```

9) que uso tiene el metodo notify_all() de una condition variable? Ejemplifique mediante código.

notify_all() es un metodo que sirve para avisarle a todos los threads que estan esperando a la condition variable que la condicion ha cambiado y puede seguir utilizando su ejecucion. Esto es util cuando varios threads estan esperando a que se cumpla una condicion y queremos despertarlos a todos para que puedan continuar.

```C++
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>

std::mutex mtx;
std::condition_variable cv;
bool ready = false;

void worker_thread(int id) {
    std::unique_lock<std::mutex> lock(mtx);
    cv.wait(lock, []{ return ready; });
    std::cout << "Worker " << id << " is executing!" << std::endl;
}

void publisher_thread() {
    {
        std::lock_guard<std::mutex> lock(mtx);
        ready = true; // Update predicate
    }
    cv.notify_all(); // Wake up all waiting threads
}
int main() {
    std::thread workers[5];
    for (int i = 0; i < 5; ++i) {
        workers[i] = std::thread(worker_thread, i);
    }
    std::thread publisher(publisher_thread);
    for (int i = 0; i < 5; ++i) {
        workers[i].join();
    }
    publisher.join();
    return 0;
}
```
